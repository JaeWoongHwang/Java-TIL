FORM delete_data .
  DATA : lt_del TYPE TABLE OF ztui524_01 WITH HEADER LINE.
  " 지울 데이터를 담을 테이블을 선언
  CLEAR : lt_del, lt_del[], gs_row, gt_row.

  CALL METHOD go_grid->get_selected_rows
    IMPORTING
      et_index_rows = gt_row.

  IF gt_row IS INITIAL.
    MESSAGE s000 WITH 'Please Select Row' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  LOOP AT gt_row INTO gs_row.
    READ TABLE gt_data INDEX gs_row-index.

    IF sy-subrc EQ 0.
      MOVE-CORRESPONDING gt_data TO lt_del.

      APPEND lt_del.
      CLEAR lt_del.
    ENDIF.
  ENDLOOP.

  IF lt_del[] IS NOT INITIAL.
    DELETE ztui524_01 FROM TABLE lt_del.
    " delete는 삭제가 목적이기 때문에 키만 있으면 됨.
    IF sy-subrc EQ 0.
      COMMIT WORK AND WAIT.
      MESSAGE s000 WITH 'Delete Success'.
      PERFORM get_data.
    ELSE.
      ROLLBACK WORK.
      MESSAGE s000 WITH 'Delete Fail' DISPLAY LIKE 'E'.
    ENDIF.
  ENDIF.
ENDFORM.

FORM get_company  USING p_fieldname.

  DATA : BEGIN OF lt_bukrs OCCURS 0,
           bukrs LIKE t001-bukrs,
           butxt LIKE t001-butxt,
         END OF lt_bukrs.

  CLEAR : lt_bukrs, lt_bukrs[].

  SELECT bukrs butxt
    INTO CORRESPONDING FIELDS OF TABLE lt_bukrs
    FROM t001.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'No Company data found'.
    EXIT.
  ENDIF.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'BUKRS'
      "return field를 의미
      dynpprog        = sy-repid

      value_tab       = lt_bukrs[]
    EXCEPTIONS

      no_values_found = 2
      OTHERS          = 3.

* Implement suitable error handling here
  ENDIF.
