*  MESSAGE i000 WITH 'Second Tab Button Click'.

    WHEN 'GET'.
      PERFORM get_edit.
    WHEN 'SET'.
      PERFORM set_edit.

      PERFORM save_edit.
    WHEN 'LOAD'.
      PERFORM load_edit.
  ENDCASE.
