FORM get_scarr .
  CLEAR : gt_scarr, gt_scarr[].
  SELECT carrid carrname currcode url
    INTO CORRESPONDING FIELDS OF TABLE gt_scarr
    FROM scarr
    WHERE carrid EQ gv_carrid1.

  IF sy-subrc NE 0.
    MESSAGE s000 WITH 'Data not found' DISPLAY LIKE 'E'.
  ENDIF.

ENDFORM.
